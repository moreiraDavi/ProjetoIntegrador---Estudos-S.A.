create table Preferencia_turno (
	id_turno numeric(1) primary key,
    nome_turno varchar(20)
);

create table Concurso (
	id_concurso numeric(5) primary key,
    nome_concurso varchar(60) unique not null,
    orgao varchar(60) not null,
    data_inscricao_inicio date not null,
    data_inscricao_final date not null,
    data_prova date not null,
    estado varchar(2) not null,
    num_vagas numeric(8) not null,
    banca varchar(30) not null,
    salario numeric(10,2) not null
);

create table Usuario (
	id_usuario numeric(5) primary key,
    nome_usuario varchar(40) not null,
    data_de_nascimento date not null,
    concurso_preferencia numeric(5),
	foreign key (concurso_preferencia) references Concurso(id_concurso),
    estado varchar(2) not null,
    dias_de_estudo numeric(1) not null,
    preferencia_turno numeric(1),
    foreign key (preferencia_turno) references Preferencia_turno(id_turno),
    email varchar(40) unique not null,
    senha varchar(200) not null
);

create table Perfil_de_estudo (
	id_perfil numeric(5) primary key,
    id_usuario numeric(5),
    foreign key (id_usuario) references Usuario(id_usuario)
		on delete cascade
        on update cascade,
	id_concurso numeric(5),
    foreign key (id_concurso) references Concurso(id_concurso)
		on delete cascade
        on update cascade
);

create table Rotina (
	id_rotina numeric(5) primary key,
    id_perfil numeric(5),
    foreign key (id_perfil) references Perfil_de_estudo(id_perfil),
    horario_segunda varchar(50),
    conteudo_segunda varchar(100),
    horario_terca varchar(50),
    conteudo_terca varchar(100),
    horario_quarta varchar(50),
    conteudo_quarta varchar(100),
    horario_quinta varchar(50),
    conteudo_quinta varchar(100),
    horario_sexta varchar(50),
	conteudo_sexta varchar(100),
    numero_semana numeric(2) not null
);
